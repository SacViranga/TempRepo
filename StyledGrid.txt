import React, { useState } from 'react';
import { DataGridPro } from '@mui/x-data-grid-pro';

const initialRows = [
  { id: 1, col1: 'Hello', col2: 'World' },
  { id: 2, col1: 'DataGridPro', col2: 'is Awesome' },
  { id: 3, col1: 'Material-UI', col2: 'rocks' },
];

const columns = [
  { field: 'col1', headerName: 'Column 1', width: 150, editable: true },
  { field: 'col2', headerName: 'Column 2', width: 150, editable: true },
];

function App() {
  const [rows, setRows] = useState(initialRows);

  const handleCellEditCommit = (params) => {
    const updatedRows = rows.map((row) =>
      row.id === params.id ? { ...row, [params.field]: params.value } : row
    );
    setRows(updatedRows);
  };

  return (
    <div style={{ height: 300, width: '100%' }}>
      <DataGridPro
        rows={rows}
        columns={columns}
        onCellEditCommit={handleCellEditCommit}
        experimentalFeatures={{ newEditingApi: true }}
        disableColumnMenu
        onCellClick={(params, event) => {
          // Check if the event was a click and not a double click or another type of interaction
          if (event.detail === 1) {
            params.api.setCellMode(params.id, params.field, 'edit');
          }
        }}
      />
    </div>
  );
}

export default App;
